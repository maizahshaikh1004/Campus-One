approve_attendance_correction

CREATE DEFINER=`root`@`localhost` PROCEDURE `approve_attendance_correction`(
     IN p_request_id INT,
     IN p_admin_id INT
 )
 BEGIN
     DECLARE v_attendance_id INT;
     DECLARE v_new_status BOOLEAN;
 
     -- Ensure admin
     IF NOT EXISTS (
         SELECT 1 FROM users
         WHERE user_id = p_admin_id AND role = 'ADMIN'
     ) THEN
         SIGNAL SQLSTATE '45000'
         SET MESSAGE_TEXT = 'Only admin can approve corrections';
     END IF;
 
     -- Fetch request details
     SELECT attendance_id, requested_status
     INTO v_attendance_id, v_new_status
     FROM attendance_correction_requests
     WHERE request_id = p_request_id
       AND status = 'PENDING';
 
     IF v_attendance_id IS NULL THEN
         SIGNAL SQLSTATE '45000'
         SET MESSAGE_TEXT = 'Invalid or already processed request';
     END IF;
 
     -- ✅ ALLOW UPDATE (MATCHES TRIGGER)
     SET @allow_attendance_update := TRUE;
 
     UPDATE attendance
     SET status = v_new_status
     WHERE attendance_id = v_attendance_id;
 
     SET @allow_attendance_update := NULL;
 
     -- Mark request approved
     UPDATE attendance_correction_requests
     SET status = 'APPROVED'
     WHERE request_id = p_request_id;
 
 END

approve_registration_request

CREATE DEFINER=`root`@`localhost` PROCEDURE `approve_registration_request`(
     IN p_request_id INT
 )
 BEGIN
     DECLARE v_name VARCHAR(100);
     DECLARE v_email VARCHAR(100);
     DECLARE v_role ENUM('FACULTY','STUDENT');
     DECLARE v_department VARCHAR(50);
 
     SELECT name, email, role, department
     INTO v_name, v_email, v_role, v_department
     FROM registration_requests
     WHERE request_id = p_request_id AND status = 'PENDING';
 
     INSERT INTO users (name, email, role, department, is_active)
     VALUES (v_name, v_email, v_role, v_department, TRUE);
 
     UPDATE registration_requests
     SET status = 'APPROVED'
     WHERE request_id = p_request_id;
 END

assign_student_coordinator

CREATE DEFINER=`root`@`localhost` PROCEDURE `assign_student_coordinator`(
     IN p_event_id INT,
     IN p_student_id INT
 )
 BEGIN
     -- Limit to max 2 coordinators per event
     IF (SELECT COUNT(*) FROM event_coordinators WHERE event_id = p_event_id) >= 2 THEN
         SIGNAL SQLSTATE '45000'
         SET MESSAGE_TEXT = 'Maximum 2 coordinators allowed per event';
     ELSE
         INSERT INTO event_coordinators (event_id, student_id)
         VALUES (p_event_id, p_student_id);
     END IF;
 END

submit_assignment

CREATE DEFINER=`root`@`localhost` PROCEDURE `submit_assignment`(
     IN p_assignment_id INT,
     IN p_student_id INT,
     IN p_file_path VARCHAR(255)
 )
 BEGIN
     DECLARE v_due_date DATE;
     DECLARE v_is_late BOOLEAN;
 
     -- Fetch due date
     SELECT due_date INTO v_due_date
     FROM assignments
     WHERE assignment_id = p_assignment_id;
 
     -- Determine late submission
     IF CURRENT_DATE > v_due_date THEN
         SET v_is_late = TRUE;
     ELSE
         SET v_is_late = FALSE;
     END IF;
 
     INSERT INTO assignment_submissions
     (assignment_id, student_id, file_path, submitted_at, is_late)
     VALUES
     (p_assignment_id, p_student_id, p_file_path, CURRENT_TIMESTAMP, v_is_late);
 END

     END IF;
 END

calculate_attendance_percentage

CREATE DEFINER=`root`@`localhost` FUNCTION `calculate_attendance_percentage`(
     p_student_id INT,
     p_subject_id INT
 ) RETURNS decimal(5,2)
     DETERMINISTIC
 BEGIN
     DECLARE total INT;
     DECLARE present INT;
 
     SELECT COUNT(*) INTO total
     FROM attendance
     WHERE student_id = p_student_id
       AND subject_id = p_subject_id;
 
     SELECT COUNT(*) INTO present
     FROM attendance
     WHERE student_id = p_student_id
       AND subject_id = p_subject_id
       AND status = TRUE;
 
     IF total = 0 THEN
         RETURN 0;
     END IF;
 
     RETURN (present / total) * 100;
 END


check_attendance_shortage

CREATE DEFINER=`root`@`localhost` FUNCTION `check_attendance_shortage`(
     p_student_id INT,
     p_subject_id INT
 ) RETURNS tinyint(1)
     DETERMINISTIC
 BEGIN
     DECLARE percentage DECIMAL(5,2);
 
     SET percentage = calculate_attendance_percentage(p_student_id, p_subject_id);
 
     IF percentage < 75 THEN
         RETURN TRUE;
     ELSE
         RETURN FALSE;
     END IF;
 END

is_late_submission

CREATE DEFINER=`root`@`localhost` FUNCTION `is_late_submission`(
     p_assignment_id INT,
     p_submission_time TIMESTAMP
 ) RETURNS tinyint(1)
     DETERMINISTIC
 BEGIN
     DECLARE v_due_date DATE;
 
     SELECT due_date INTO v_due_date
     FROM assignments
     WHERE assignment_id = p_assignment_id;
 
     IF DATE(p_submission_time) > v_due_date THEN
         RETURN TRUE;
     ELSE
         RETURN FALSE;
     END IF;
 END
